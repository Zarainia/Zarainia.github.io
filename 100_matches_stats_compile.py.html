23   		filtered_stat_losses = sum (process_function (personal_stats[match][stat_index]) for match in filtered_losses if personal_stats[match])
24   		filtered_stat_close = sum (process_function (personal_stats[match][stat_index]) for match in filtered_close if personal_stats[match])
25   		
26   		if total:
27   			table.add_row ([f"total {stat_name}", filtered_stat_total])
28   		
29   		table.add_row ([f"mean {stat_name} per match", filtered_stat_total / total_stats_matches])
30   		table.add_row ([f"mean {stat_name} (win)", filtered_stat_wins / total_stats_wins])
31   		table.add_row ([f"mean {stat_name} (loss)", filtered_stat_losses / total_stats_losses])
32   		try:
33   			table.add_row ([f"mean {stat_name} (close)", filtered_stat_close / total_stats_close])
34   		except ZeroDivisionError:
35   			pass
36   		table.add_row ([f"highest {stat_name}", max(process_function(personal_stats[match][stat_index]) for match in filtered_matches if personal_stats[match])])
37   		table.add_row ([f"highest {stat_name} (win)", max(process_function(personal_stats[match][stat_index]) for match in filtered_wins if personal_stats[match])])
38   		table.add_row ([f"highest {stat_name} (loss)", max(process_function(personal_stats[match][stat_index]) for match in filtered_losses if personal_stats[match])])
39   		try:
40   			table.add_row ([f"highest {stat_name} (close)", max(process_function(personal_stats[match][stat_index]) for match in filtered_close if personal_stats[match])])
41   		except ValueError:
42   			pass
43   		table.add_row ([f"lowest {stat_name}", min(process_function(personal_stats[match][stat_index]) for match in filtered_matches if personal_stats[match])])
44   		table.add_row ([f"lowest {stat_name} (win)", min(process_function(personal_stats[match][stat_index]) for match in filtered_wins if personal_stats[match])])
45   		table.add_row ([f"lowest {stat_name} (loss)", min(process_function(personal_stats[match][stat_index]) for match in filtered_losses if personal_stats[match])])
46   		try:
47   			table.add_row ([f"lowest {stat_name} (close)", min(process_function(personal_stats[match][stat_index]) for match in filtered_close if personal_stats[match])])
48   		except ValueError:
49   			pass
50   		return filtered_stat_total, filtered_stat_wins, filtered_stat_losses, filtered_stat_close

53   	def add_scores(matches, title):
54   		try:
55   			table.add_row(["mean team score" + title, sum(matches_scores[match][0] for match in matches) / len(matches)])
56   			table.add_row(["mean enemy score" + title, sum(matches_scores[match][1] for match in matches) / len(matches)])
57   			table.add_row(["highest team score" + title, max(matches_scores[match][0] for match in matches)])
58   			table.add_row(["lowest team score" + title, min(matches_scores[match][0] for match in matches)])
59   			table.add_row(["highest enemy score" + title, max(matches_scores[match][1] for match in matches)])
60   			table.add_row(["lowest enemy score" + title, min(matches_scores[match][1] for match in matches)])
61   		except ZeroDivisionError:
62   			pass
63   	
64   	table = prettytable.PrettyTable (["field", "value"])
65   	table.set_style (prettytable.PLAIN_COLUMNS)
66   	table.align = "l"
67   	table.header = False
68   	table.float_format = ".2"
69   	table.title = name
70   	
71   	table.add_row (["matches", len (filtered_matches)])
72   	
73   	if ships:
74   		table.add_row (["mangler matches", sum (1 for match in filtered_matches if "GSS-3 Mangler" in matches_dict[match]["snippet"]["description"])])
75   		table.add_row (["sting matches", sum (1 for match in filtered_matches if "S-13 Sting" in matches_dict[match]["snippet"]["description"])])
76   		
77   	filtered_wins = filtered_matches.intersection (win_videos)
78   	table.add_row (["wins", len (filtered_wins)])
79   	filtered_losses = filtered_matches.intersection (loss_videos)
80   	table.add_row (["losses", len (filtered_losses)])
81   	table.add_row (["win rate", f"{len (filtered_wins) / len (filtered_matches):.2%}"])
82   	filtered_close = filtered_matches.intersection(close_videos)
83   	table.add_row(["close matches", len(filtered_close)])
84   	table.add_row(["percent close", f"{len (filtered_close) / len (filtered_matches):.2%}"])
85   	
86   	if scores:
87   		add_scores(filtered_matches, "")
88   		add_scores(filtered_wins, " (win)")
89   		add_scores(filtered_losses, " (loss)")
90   		add_scores(filtered_close, " (close)")
91   	
92   	total_stats_matches = sum(1 for match in filtered_matches if personal_stats[match])
93   	total_stats_wins = sum(1 for match in filtered_wins if personal_stats[match])
94   	total_stats_losses = sum(1 for match in filtered_losses if personal_stats[match])
95   	total_stats_close = sum(1 for match in filtered_close if personal_stats[match])
96   	filtered_kills, filtered_kills_wins, filtered_kills_losses, filtered_kills_close = add_partial_stat("kills", 1)
97   	add_partial_stat("assists", 2)
98   	filtered_deaths, filtered_deaths_wins, filtered_deaths_losses, filtered_deaths_close = add_partial_stat("deaths", 3)
99   	
100  	table.add_row(["KDR", filtered_kills / filtered_deaths])
101  	table.add_row(["KDR (win)", filtered_kills_wins / filtered_deaths_wins])
102  	table.add_row(["KDR (loss)", filtered_kills_losses / filtered_deaths_losses])
103  	try:
104  		table.add_row(["KDR (close)", filtered_kills_close / filtered_deaths_close])
105  	except ZeroDivisionError:
106  		pass
107  	
108  	add_partial_stat("damage", 4)
109  	add_partial_stat("hit rate", 5, total=False)
110  	add_partial_stat("medals", 6, process_function=lambda x: int(x.replace('x', '')))
111  	
112  	return table

124  	if ylabel:

126  	plt.savefig (f"{character_name}_{filename}.png")

146  def create_area_plot(x, y1, y2, title, filename, xlabel="date", ylabel=None):
147  	axis = plt.subplot()
148  	axis.xaxis.set_major_formatter (matplotlib.dates.DateFormatter (DATE_FORMAT))
149  	axis.xaxis.set_major_locator (matplotlib.dates.DayLocator ())
150  	axis.yaxis.set_major_formatter (matplotlib.dates.DateFormatter (TIME_FORMAT))
151  	axis.yaxis.set_major_locator (matplotlib.dates.MinuteLocator (interval=15))
152  	plt.plot(x, y1,  color="blue")
153  	plt.plot(x, y2,  color="blue")
154  	axis.fill_between (x, y1, y2, alpha=0.2, color="blue")
155  	fig = plt.gcf ()
156  	fig.autofmt_xdate ()
157  	fig.set_size_inches (20, 10)
158  	plt.title(title)
159  	if xlabel:
160  		plt.xlabel(xlabel)
161  	if ylabel:
162  		plt.ylabel(ylabel)
163  	plt.savefig (f"{character_name}_{filename}.png")
164  	plt.close()
